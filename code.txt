    public static <T extends Comparable<T>> T[] insertionSort(T[] arr) {
        int length = arr.length;
        for (int i = 0; i < length; i++) {
            int j = i;
            while (j > 0 && arr[j].compareTo(arr[j - 1]) < 0) {
                T temp = arr[j - 1];
                arr[j - 1] = arr[j];
                arr[j] = temp;
                j--;
            }
        }
        return arr;
    }
